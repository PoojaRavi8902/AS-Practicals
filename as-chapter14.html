import java.util.Scanner;

public class MonthDays {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter month number: ");
        int month = input.nextInt();
        System.out.print("Enter year: ");
        int year = input.nextInt();
        int days;

        switch(month) {
            case 1: case 3: case 5: case 7: case 8: case 10: case 12:
                days = 31; break;
            case 4: case 6: case 9: case 11:
                days = 30; break;
            case 2:
                if ((year % 400 == 0) || (year % 4 == 0 && year % 100 != 0))
                    days = 29;
                else
                    days = 28;
                break;
            default:
                System.out.println("Invalid month number");
                return;
        }
        System.out.println("Number of days: " + days);
    }
}

Write a function to implement the pseudocode:

FUNCTION Factorial(Number : INTEGER) RETURNS INTEGER
    DECLARE Product : INTEGER
    Product ← 1
    FOR n ← 2 TO Number
        Product ← Product * n
    NEXT n
    RETURN Product
ENDFUNCTION

Java implementation:

int factorial(int number) {
    int product = 1;
    for (int n = 2; n <= number; n++) {
        product *= n;
    }
    return product;
}

Passing parameters to procedures (by value):

void outputSymbols(int numberOfSymbols, char symbol) {
    for (int count = 1; count <= numberOfSymbols; count++) {
        System.out.print(symbol);
    }
    System.out.println();
}

Passing parameters by reference:

// Java does not support passing primitive types by reference
// Arrays and objects can be passed by reference

void adjustValuesForNextRow(int[] values) { // values[0] = Spaces, values[1] = Symbols
    values[0] = values[0] - 1;
    values[1] = values[1] + 2;

// rowValues[0] and rowValues[1] are updated

TIP:

  • To return multiple values from a Java subroutine, group them in an array or create a class and return an object.
  • If only one value needs to be returned, write it as a function.
  • Preferably, write multiple functions for clarity rather than returning complex objects.

14.13 Putting it all together – Pyramid Program in Java

The pyramid-drawing program demonstrates passing parameters, local vs main program variables, and using subroutines.

package ex1;
import java.util.Scanner;

public class Ex1 {

    static final char SPACE = ' ';

    // Get symbol from user
    public static char getSymbol() {
        Scanner console = new Scanner(System.in);
        System.out.print("What symbol do you want to use? ");
        String response = console.next();
        return response.charAt(0);
    }

    // Input max number of symbols (must be odd)
    public static int inputMaxNumberOfSymbols() {
        Scanner console = new Scanner(System.in);
        int number = 0;
        while ((number % 2) == 0) {
            System.out.print("How many symbols make the base? ");
            number = console.nextInt();
        }
        return number;
    }

    // Output characters (spaces or symbols)
    public static void outputChars(int number, char symbol) {
        for (int count = 0; count < number; count++) {
            System.out.print(symbol);
        }
    }

    // Adjust number of spaces for next row
    public static int adjustSpacesForNextRow(int spaces) {
        spaces--;
        return spaces;
    }

    // Adjust number of symbols for next row
    public static int adjustSymbolsForNextRow(int symbols) {
        symbols = symbols + 2;
        return symbols;
    }

    public static void main(String[] args) {
        char thisSymbol = getSymbol();
        int maxNumberOfSymbols = inputMaxNumberOfSymbols();
        int numberOfSpaces = (maxNumberOfSymbols + 1) / 2;
        int numberOfSymbols = 1;

        while (numberOfSymbols <= maxNumberOfSymbols) {
            outputChars(numberOfSpaces, SPACE);
            outputChars(numberOfSymbols, thisSymbol);
            System.out.println(); // move to new line
            numberOfSpaces = adjustSpacesForNextRow(numberOfSpaces);
            numberOfSymbols = adjustSymbolsForNextRow(numberOfSymbols);
        }
    }
}

Discussion Point:

  • The procedures OutputSpaces and OutputSymbols from earlier examples have been replaced by a single procedure outputChars without changing the program’s effect.
  • The identifiers used inside the subroutines are independent of the main program’s variable names, showing the distinction between parameters/local variables and main program variables.
  • Passing parameters by value (for primitives) ensures changes inside a subroutine do not affect the original variable unless returned.

14.14 Arrays

Arrays are used to store multiple values of the same type in a single variable. They can be one-dimensional (1D) or two-dimensional (2D).

Creating 1D Arrays

In Java, a 1D array can be declared and initialized as follows:

// 1D array declaration and initialization
String[] list1 = {"Fred", "Jack", "Ali"}; // 3 elements
int[] list2 = new int[6];                 // 6 elements, default 0
int[] list3 = new int[100];               // 100 elements, default 0
String[] aList = new String[26];          // 26 elements, default null

You can access or modify elements using an index (0-based):

// Accessing and modifying elements
list2[0] = 10;
list2[1] = 20;
list3[99] = 500;
aList[3] = "D";

To print the contents of an array, use a loop:

// Printing 1D arrays
System.out.println("1D array list1:");
for (int i = 0; i < list1.length; i++) {
    System.out.print(list1[i] + " ");
}
System.out.println();

Creating 2D Arrays

2D arrays represent data in rows and columns. Declaration and initialization in Java:

// 2D array (6 rows x 7 columns)
int[][] board = new int[6][7];

// Assigning values
board[0][0] = 1;
board[5][6] = 2;
board[2][3] = 5;

Access elements using two indices (row and column, 0-based):

// Accessing a specific element
board[3][4] = 0;

Print the 2D array:

// Printing 2D array
System.out.println("2D array board:");
for (int i = 0; i < board.length; i++) {
    for (int j = 0; j < board[i].length; j++) {
        System.out.print(board[i][j] + " ");
    }
    System.out.println();
}

Dynamic Filling Example

Fill the array with a calculation (row + column):

// Dynamic filling
for (int i = 0; i < board.length; i++) {
    for (int j = 0; j < board[i].length; j++) {
        board[i][j] = i + j;
    }
}

14.15 Text Files

Writing to a Text File

In Java, you can write to a text file using FileWriter and PrintWriter. If the file already exists, it will be overwritten unless you open it in append mode.


import java.io.FileWriter;
import java.io.PrintWriter;
import java.io.IOException;

public class WriteTextFileExample {
    public static void main(String[] args) throws IOException {
        // Create a FileWriter (false = overwrite)
        FileWriter fileHandle = new FileWriter("SampleFile.TXT", false);
        PrintWriter printLine = new PrintWriter(fileHandle);

        // Text to write
        String lineOfText = "Hello, this is a sample text line.";

        // Write line to file
        printLine.printf("%s%n", lineOfText);

        // Close the file
        printLine.close();
    }
}

Reading from a Text File

To read from an existing text file in Java, use FileReader and BufferedReader. Reading stops when the end of the file is reached.


import java.io.FileReader;
import java.io.BufferedReader;
import java.io.IOException;

public class ReadTextFileExample {
    public static void main(String[] args) throws IOException {
        FileReader fileHandle = new FileReader("SampleFile.TXT");
        BufferedReader textReader = new BufferedReader(fileHandle);

        String lineOfText = textReader.readLine();
        while (lineOfText != null) {
            System.out.println(lineOfText);
            lineOfText = textReader.readLine();
        }

        textReader.close();
    }
}

Appending to a Text File

To append text to an existing file, pass true as the second parameter to FileWriter.


import java.io.FileWriter;
import java.io.PrintWriter;
import java.io.IOException;

public class AppendTextFileExample {
    public static void main(String[] args) throws IOException {
        // Open file in append mode
        FileWriter fileHandle = new FileWriter("SampleFile.TXT", true);
        PrintWriter printLine = new PrintWriter(fileHandle);

        // Text to append
        String lineOfText = "This line will be appended.";

        printLine.printf("%s%n", lineOfText);
        printLine.close();
    }
}

Example Task: Tally Chart

Fred wants to tally student hobbies. Each hobby is represented by an index in an array.


import java.util.Scanner;

public class HobbyTally {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        // Initialize tally array for 5 hobbies
        int[] tally = new int[6]; // index 0 unused

        int choice;
        System.out.println("Enter hobby choices (1-5, 0 to end):");
        do {
            choice = sc.nextInt(); // Input hobby number
            if (choice >= 1 && choice <= 5) {
                tally[choice]++; // Increment tally
            }
        } while (choice != 0);

        // Output total for each hobby
        System.out.println("\nTally of hobbies:");
        for (int i = 1; i <= 5; i++) {
            System.out.println("Hobby " + i + ": " + tally[i]);
        }

        sc.close();
    }
}

// Pseudocode
FUNCTION IsValidRegistration(reg)
    IF LENGTH(reg) ≠ 7 THEN
        RETURN FALSE
    ENDIF

    // Pattern 1: LLNNLLL
    IF ISUPPER(reg[1..2]) AND ISDIGIT(reg[3..4]) AND ISUPPER(reg[5..7]) THEN
        RETURN TRUE
    ENDIF

    // Pattern 2: LLLNNLL
    IF ISUPPER(reg[1..3]) AND ISDIGIT(reg[4..5]) AND ISUPPER(reg[6..7]) THEN
        RETURN TRUE
    ENDIF

    RETURN FALSE
ENDFUNCTION
